    return a + b

def faktoriyel(n):
    if n == 0:
        return 1
    else:
        return n * faktoriyel(n-1)

# JavaScript kodu örnekleri
function topla(a, b) {
    return a + b;
}

function faktoriyel(n) {
    if (n === 0) {
        return 1;
    } else {
        return n * faktoriyel(n - 1);
    }
}
